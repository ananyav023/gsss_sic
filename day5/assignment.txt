#Accept a number from user and find the next possible smallest number which is bigger than the given number having all the digits of the given number

n = input("Enter a number: ")
digits = list(n)
i = len(digits) - 2
while i >= 0 and digits[i] >= digits[i+1]:
    i -= 1

if i == -1:
    print("Not possible")
else:
    j = len(digits) - 1
    while digits[j] <= digits[i]:
        j -= 1

    digits[i], digits[j] = digits[j], digits[i]

    digits[i+1:] = reversed(digits[i+1:])

    result = ''.join(digits)
    print("Next smallest bigger number:", result)
